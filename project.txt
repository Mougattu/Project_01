**Shell Script** (`install_tomcat.sh`)
This makes it runnable in one go:

```bash
#!/bin/bash
# Install Java
sudo yum update -y
sudo yum install java-1.8.0-openjdk -y

# Install Tomcat
cd /opt
sudo wget https://downloads.apache.org/tomcat/tomcat-9/v9.0.108/bin/apache-tomcat-9.0.108.tar.gz
sudo tar xvf apache-tomcat-9.0.108.tar.gz
sudo mv apache-tomcat-9.0.108 tomcat9
sudo chmod -R 755 /opt/tomcat9
sudo chmod +x /opt/tomcat9/bin/*.sh
sudo chown -R ec2-user:ec2-user /opt/tomcat9

# Start Tomcat
/opt/tomcat9/bin/startup.sh

echo "Tomcat installed and started successfully."
echo "Deploy your WAR file by copying it to /opt/tomcat9/webapps/"
